\begin{frame}{Three Layers Of Meaning}
\textbf{I think the recent work only makes sense if we separate three claims}

\vspace{0.6em}
\begin{columns}[T]
\column{0.32\textwidth}
\begin{block}{1. Representation Meaning}
\small
The latent or state contains real developmental information.

\vspace{0.4em}
Typical evidence:
\begin{itemize}
    \item developmental time
    \item founder
    \item depth
    \item future structure probes
\end{itemize}
\end{block}

\column{0.32\textwidth}
\begin{block}{2. Predictive Meaning}
\small
The \emph{predicted} future object still carries true future structure.

\vspace{0.4em}
This is stronger than representation quality:
\begin{itemize}
    \item not just readable true states
    \item but readable predicted future states
\end{itemize}
\end{block}

\column{0.32\textwidth}
\begin{block}{3. Biological Meaning}
\small
The model supports mechanism-level biological interpretation.

\vspace{0.4em}
This would require:
\begin{itemize}
    \item stronger causal/perturbational evidence
    \item stable biological coordinates
    \item more than coarse readout success
\end{itemize}
\end{block}
\end{columns}
\end{frame}

\begin{frame}{Where We Are In Those Three Layers}
\textbf{Current position}
\begin{itemize}
    \item \textbf{Representation meaning: established.}
    Broad patch and embryo latents clearly carry developmental structure.
    \item \textbf{Predictive meaning: partially established.}
    Reconstruction-backed embryo future-set models are near the right regime,
    but predicted future structure is still uneven.
    \item \textbf{Biological meaning: not yet established.}
    We should not claim mechanism-level biology from the current models.
\end{itemize}

\vspace{0.7em}
\textbf{So the real question today is not}
\begin{itemize}
    \item ``did we learn something?''
\end{itemize}

\textbf{It is}
\begin{itemize}
    \item ``what kind of predictive object can support layer 2 without pretending we already have layer 3?''
\end{itemize}
\end{frame}

\begin{frame}[shrink=5]{Project Goal And Current Reframing}
\textbf{Long-term goal}
\begin{itemize}
    \item Predict whole-organism developmental progression.
    \item Keep transcriptomic state as the main biological variable.
\end{itemize}

\vspace{0.5em}
\textbf{What changed}
\begin{itemize}
    \item I no longer trust ``predict a pooled embryo latent and decode biology later'' as the cleanest route.
    \item That object was overloaded:
    observation summary, predictive state, readout space, and compression.
\end{itemize}

\vspace{0.3em}
\textbf{Active simplification}
\begin{center}
\small
\textbf{history gene tokens $\rightarrow$ future clean gene tokens}\\
instead of current latent $\rightarrow$ future latent $\rightarrow$ biology
\end{center}

\vspace{0.3em}
\textbf{Meeting thesis}
\begin{itemize}
    \item The pooled-latent route helped us reach layer 1.
    \item It did not give a clean layer-2 predictive object.
    \item The next move is direct gene-space prediction with longer temporal context.
\end{itemize}
\end{frame}

\begin{frame}{Data Reality}
\textbf{Raw dataset}
\begin{itemize}
    \item \texttt{dataset/processed/nema\_extended\_large2025.h5ad}
    \item 234,888 cells
    \item 27,138 genes
    \item 267 unique embryo times
\end{itemize}

\vspace{0.5em}
\textbf{But the effective predictive dataset is much smaller}
\begin{itemize}
    \item Current predictive contract uses:
    \begin{itemize}
        \item \texttt{dt = 20 min}
        \item \texttt{time window = 10 min}
        \item \texttt{min cells per window = 32}
    \end{itemize}
    \item Under this contract:
    \begin{itemize}
        \item patch dataset: 31 usable time pairs total
        \item train/val split: 25 / 6 pairs
        \item temporal history with 4 patches: 24 total, 20 / 4 train/val pairs
    \end{itemize}
    \item So the true bottleneck is \textbf{independent time transitions}, not raw cell count.
\end{itemize}
\end{frame}

\begin{frame}{What The Data Support}
\textbf{What is justified}
\begin{itemize}
    \item Longer sequence context
    \item More historical patches
    \item Moderate transformer scale
    \item Direct gene-space prediction
\end{itemize}

\vspace{0.5em}
\textbf{What is not justified}
\begin{itemize}
    \item Large-capacity latent world models
    \item Heavy gauge-fixing machinery on top of weak time-pair supervision
    \item Overly complicated auxiliary heads that assume more temporal diversity than the data provide
\end{itemize}

\vspace{0.5em}
\textbf{Practical reading}
\begin{itemize}
    \item Increase sequence length before increasing width/depth.
    \item Bias toward simpler objectives in biologically meaningful coordinates.
\end{itemize}
\end{frame}

\textbf{The current best compromise is still the reconstruction-led \texttt{spatialtuned} future-set checkpoint}

\vspace{0.5em}
\small
\begin{tabular}{lrrrrr}
\toprule
\textbf{Checkpoint} & founder & celltype & depth & spatial & split \\
\midrule
\texttt{spatialtuned} & 0.159 & 0.450 & 0.554 & -1.019 & -1.211 \\
\bottomrule
\end{tabular}

\vspace{0.8em}
\textbf{Why this matters}
\begin{itemize}
    \item It is not the final answer.
    \item But it is the best tradeoff we found:
    \begin{itemize}
        \item semantic/developmental structure remained readable,
        \item weak spatial tether helped a bit,
        \item more complicated repairs usually made things worse.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{The Core Failure Pattern}
\textbf{Repeated latent-space failure mode}
\begin{itemize}
    \item Predicted future latent often contained information.
    \item But it did \textbf{not} live in the same readable coordinate system as true future latent.
    \item Result:
    \begin{itemize}
        \item self-readability could be strong,
        \item transfer from probes trained in true space remained weak.
    \end{itemize}
\end{itemize}

\vspace{0.5em}
\textbf{This is the right interpretation}
\begin{itemize}
    \item not ``the model learned nothing'',
    \item but ``the model learned a future representation with the wrong gauge''.
\end{itemize}

\vspace{0.5em}
\textbf{Why this matters}
\begin{itemize}
    \item It is a standard world-model state-space problem, not a biology-specific oddity.
    \item It means better losses or auxiliary heads do not automatically fix the real issue.
\end{itemize}
\end{frame}

\begin{frame}{What The Diagnostics Actually Showed}
\textbf{Oracle and alignment analyses changed the interpretation}
\begin{itemize}
    \item Readout was not the main problem.
    \item Object support or matching was not the main short-term problem either.
    \item The dominant failure looked like a largely \textbf{global linear misalignment}.
\end{itemize}

\vspace{0.5em}
\textbf{Representative pattern}
\begin{itemize}
    \item direct transfer from true-space probes to predicted space: poor
    \item ridge-aligned transfer: almost fully recovers
\end{itemize}

\vspace{0.5em}
\textbf{Implication}
\begin{itemize}
    \item The project was spending too much effort repairing a pooled latent object that was already conceptually overloaded.
    \item This suggested a more radical simplification:
    avoid pooled latent as the main predictive state.
\end{itemize}
\end{frame}

\begin{frame}{Why I No Longer Trust Pooled Latent As The Main State}
\textbf{Pooled latent was doing too many incompatible things at once}
\begin{itemize}
    \item observation summary
    \item predictive world state
    \item biological readout space
    \item compression bottleneck
\end{itemize}

\vspace{0.5em}
\textbf{Consequences}
\begin{itemize}
    \item founder/celltype/depth were often easier to preserve
    \item spatial and split were repeatedly the first things to break
    \item more flexible pooling or alignment heads usually made the gauge problem worse
\end{itemize}

\vspace{0.5em}
\textbf{Current position}
\begin{itemize}
    \item Pooled latent may remain a diagnostic or downstream summary.
    \item It should no longer define the main training contract.
\end{itemize}
\end{frame}

\begin{frame}{Gene Is Primary; Celltype Is Too Coarse}
\textbf{This is the biological correction that matters most}
\begin{itemize}
    \item Gene expression is the first-order developmental variable in this project.
    \item Celltype is a derived summary.
    \item A model that looks good on celltype while failing on gene is not scientifically satisfying.
\end{itemize}

\vspace{0.5em}
\textbf{What changed operationally}
\begin{itemize}
    \item I moved away from celltype-first interpretation.
    \item I rewrote evaluation toward gene-first and token/set-level thinking.
    \item This makes the latent-world-model route look less attractive, because gene space already provides a fixed biological coordinate system.
\end{itemize}
\end{frame}

\begin{frame}{Why JiT Became The Right Reference}
\textbf{The most useful lesson from JiT is conceptual}
\begin{itemize}
    \item Directly predict clean data.
    \item Do not force the model to predict noisy, displaced, or arbitrarily reparameterized targets if the observation space itself is meaningful.
\end{itemize}

\vspace{0.5em}
\textbf{Translation to this project}
\begin{itemize}
    \item image patch $\rightarrow$ gene patch / gene token set
    \item clean image target $\rightarrow$ clean future gene target
    \item tokenizer-free pixel prediction $\rightarrow$ latent-free gene prediction
\end{itemize}

\vspace{0.5em}
\textbf{In one sentence}
\begin{itemize}
    \item The right simplification is not ``a better latent''.
    \item It is ``a more direct gene-space predictive object''.
\end{itemize}
\end{frame}

\begin{frame}{Current JiT-Like Model}
\textbf{Current implementation}
\begin{itemize}
    \item Current/historical patch gene tokens enter directly.
    \item Future query tokens are learned queries.
    \item A bidirectional transformer processes both together.
    \item The model directly predicts future clean genes.
\end{itemize}

\vspace{0.5em}
\textbf{Important negative choices}
\begin{itemize}
    \item no pooled latent state
    \item no local code bottleneck
    \item no decoder tether
    \item no latent alignment head
    \item no hand-designed anchor ontology inside the model
\end{itemize}
\end{frame}

\begin{frame}{Why This Is Better Thought Of As A Video-Like Model}
\textbf{The right analogy is no longer a latent world model}
\begin{itemize}
    \item Past patches behave like history frames.
    \item Future query tokens behave like next-frame prediction queries.
    \item The model predicts clean token content directly.
\end{itemize}

\vspace{0.5em}
\textbf{Current temporal configuration}
\begin{itemize}
    \item \texttt{history\_patches = 4}
    \item \texttt{context\_size = 32}
    \item observed history tokens: \(4 \times 32 = 128\)
    \item future query tokens: \(32\)
    \item total sequence length: \(160\)
\end{itemize}

\vspace{0.5em}
\textbf{Why this is the right scale for now}
\begin{itemize}
    \item It uses the data better by expanding context.
    \item It does not assume the dataset can support a very large latent world model.
\end{itemize}
\end{frame}

\begin{frame}{What We Know From The Simple Gene Baseline}
\textbf{Current local-region baseline}
\begin{itemize}
    \item Task: current local mean gene \(\rightarrow\) future local mean gene
    \item Models: persistence, ridge, 2-layer MLP
    \item Cross-validation by time pair
\end{itemize}

\vspace{0.5em}
\textbf{Main result}
\begin{itemize}
    \item The task is not empty.
    \item A small MLP beats persistence in MSE.
    \item But independent pair count is still small enough that per-gene \(R^2\) remains unstable.
\end{itemize}

\vspace{0.5em}
\textbf{Interpretation}
\begin{itemize}
    \item Direct gene prediction is worth pursuing.
    \item We should not over-read noisy gene-level statistics from tiny pair counts.
\end{itemize}
\end{frame}

\begin{frame}{What I Now Believe}
\textbf{Main claims}
\begin{itemize}
    \item The pooled-latent world-model route taught us a real lesson, but is probably not the right mainline.
    \item The strongest remaining scientific simplification is:
    \[
    \text{history gene patches} \rightarrow \text{future clean gene patches}
    \]
    \item Data support longer sequence context better than much larger model capacity.
    \item Gene-first evaluation is more defensible than celltype-first comfort.
\end{itemize}

\vspace{0.5em}
\textbf{What I no longer want to keep doing}
\begin{itemize}
    \item repairing pooled latent with more geometry tricks
    \item inventing more auxiliary heads before fixing the predictive object
    \item treating a derived label space as stronger evidence than gene-space prediction
\end{itemize}
\end{frame}

\begin{frame}{Immediate Next Steps}
\textbf{The near-term plan is deliberately narrow}
\begin{enumerate}
    \item Train the longer-history JiT-like gene patch model properly.
    \item Compare it against the short-context version under gene-space metrics.
    \item Keep model size modest; scale sequence before scaling width/depth.
    \item Re-introduce additional structure only if direct gene prediction clearly saturates.
\end{enumerate}

\vspace{0.7em}
\textbf{What would count as convincing progress}
\begin{itemize}
    \item better future gene prediction under held-out time-pair evaluation
    \item cleaner scaling with historical context
    \item less dependence on complicated latent-space interpretation
\end{itemize}
\end{frame}

\begin{frame}{Questions For Discussion}
\textbf{What I want feedback on}
\begin{itemize}
    \item Do we agree that pooled latent should be downgraded from main state object to diagnostic summary?
    \item Is direct gene-space prediction the right simplification, or should we preserve a more explicit state bottleneck?
    \item How much history should we expose before we should worry about overfitting pair-specific structure?
    \item Should we add \textbf{BERT-style masking / denoising augmentation} on the \emph{input side}:
    \begin{itemize}
        \item token dropout,
        \item whole-history-patch dropout,
        \item partial gene-channel masking,
    \end{itemize}
    while keeping the target unchanged as future clean gene prediction?
    \item If we add structure back later, what biological structure is worth restoring first:
    \begin{itemize}
        \item spatial geometry,
        \item division/branching,
        \item or a stronger local set object?
    \end{itemize}
\end{itemize}
\end{frame}
